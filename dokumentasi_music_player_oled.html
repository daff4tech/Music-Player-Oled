Hardware

Komponen yang Dibutuhkan

🔲

ESP32-C3 Supermini

Board utama. RISC-V 160MHz, WiFi & BT bawaan, 400KB SRAM.

📺

OLED SH1106 128×64

Layar monokrom I2C. Alamat default 0x3C. Tegangan 3.3V.

🔌

Kabel Jumper

4 buah kabel female-to-female untuk koneksi VCC, GND, SDA, SCL.

Skema Wiring

Persiapan

Cara Upload ke ESP32-C3

Install Arduino IDE

Download dari arduino.cc/en/software. Gunakan versi 2.x untuk pengalaman terbaik.

Tambahkan Board ESP32

Buka File → Preferences, tambahkan URL berikut ke "Additional Board Manager URLs":

https://raw.githubusercontent.com/espressif/arduino-esp32/gh-pages/package_esp32_index.json

Kemudian buka Tools → Board → Boards Manager, cari esp32, install paket dari Espressif.

Install Library

Buka Sketch → Include Library → Manage Libraries, lalu install:

Adafruit SH110X

Driver utama untuk OLED SH1106/SH1107. Install ini — GFX ikut otomatis.

Adafruit GFX Library

Library grafis dasar untuk teks, garis, kotak. Biasanya terinstall otomatis.

Pilih Board & Port

Sambungkan ESP32-C3 via USB, lalu pilih:

Tools → Board: ESP32C3 Dev Module
Tools → USB CDC On Boot: Enabled
Tools → Port: COMx atau /dev/ttyUSB0

💡 Jika port tidak muncul, tekan tombol BOOT di board sambil colok USB, atau install driver CH340/CP2102.

Upload Kode

Buka file music_player_oled.ino, klik tombol Upload (→). Tunggu "Done uploading", OLED langsung menyala.

Referensi Kode

Dokumentasi Fungsi

setup() void

Dijalankan sekali saat board dinyalakan. Menginisialisasi I2C, OLED, nilai awal equalizer (tinggi & kecepatan acak per batang), dan mencatat waktu mulai lirik pertama.

loop() void

Loop utama yang berjalan terus-menerus. Memanggil keempat fungsi update dan gambar. Tidak menggunakan delay() — semua berbasis millis() agar semua animasi berjalan bersamaan tanpa saling memblokir.

updateLirik(sekarang) void

Membandingkan waktu saat ini dengan waktu mulai lirik + durasi lirik. Jika sudah lewat, indexLirik dinaikkan (modulo total lirik agar looping). Waktu mulai baru dicatat.

updateEqualizer(sekarang) void

Tiap batang punya target tinggi dan tinggi saat ini. Setiap iterasi, tinggi saat ini digeser mendekati target menggunakan interpolasi linear (lerp). Target baru dipilih secara acak pada interval berbeda-beda per batang, menghasilkan gerakan yang natural dan tidak bersamaan.

updateIkon(sekarang) void

Toggle boolean ikonTampil setiap 500ms. Ikon musik di pojok kanan atas akan tampil/hilang bergantian, menciptakan efek kedip.

gambarDisplay() void

Dipanggil ~25x per detik (setiap 40ms). Membersihkan buffer, lalu menggambar semua elemen secara berurutan: judul → ikon → garis pemisah → lirik → equalizer. Terakhir, display.display() mengirim buffer ke layar sekaligus.

pusatkanTeks(teks) int

Menghitung posisi X agar teks berada di tengah layar. Rumus: (128 - panjang_teks × 6) / 2. Font size 1 pada Adafruit GFX berukuran 6×8 piksel per karakter.

Source Code

Kode Lengkap

music_player_oled.ino 
// ============================================================
//  Music Player OLED - ESP32-C3 Supermini
//  Board   : ESP32-C3 Supermini
//  Display : SH1106 128x64 I2C (alamat 0x3C)
//  Library : Adafruit GFX + Adafruit SH110X
// ============================================================

#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SH110X.h>

// ─── Konfigurasi Layar ───────────────────────────────────────
#define SCREEN_WIDTH  128
#define SCREEN_HEIGHT 64
#define OLED_ADDR     0x3C
#define OLED_RESET    -1
#define SDA_PIN 6
#define SCL_PIN 7

Adafruit_SH1106G display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// ─── Struct Lirik ────────────────────────────────────────────
struct Lyric {
  const char* line1;
  const char* line2;
  unsigned long durasi;
};

const Lyric lirik[] = {
  { "tapi semua yang",   "kau cinta akan pergi", 5620 },
  { "maka tunjukkan",    "",                      2730 },
  { "sebelum terlambat", "",                      4140 },
  { "kita slalu pikir",  "kita punya waktu",      5170 },
  { "lalu kita pilih",   "untuk nanti dulu",      5900 },
};

const int TOTAL_LIRIK = sizeof(lirik) / sizeof(lirik[0]);
int indexLirik = 0;
unsigned long waktuGantiLirik = 0;

// ─── Equalizer ───────────────────────────────────────────────
#define EQ_BATANG     13
#define EQ_LEBAR       7
#define EQ_GAP         2
#define EQ_Y_BAWAH    63
#define EQ_MAX_TINGGI 12

float eqTarget[EQ_BATANG];
float eqSekarang[EQ_BATANG];
float eqKecepatan[EQ_BATANG];
unsigned long eqWaktuGanti[EQ_BATANG];

bool ikonTampil = true;
unsigned long waktuIkon = 0;
#define IKON_INTERVAL 500

unsigned long waktuUpdateDisplay = 0;
#define DISPLAY_INTERVAL 40

// ─── Setup ───────────────────────────────────────────────────
void setup() {
  Serial.begin(115200);
  Wire.begin(SDA_PIN, SCL_PIN);
  if (!display.begin(OLED_ADDR, true)) {
    while (true);
  }
  display.clearDisplay();
  display.display();
  randomSeed(analogRead(0));
  for (int i = 0; i < EQ_BATANG; i++) {
    eqSekarang[i]   = random(2, EQ_MAX_TINGGI);
    eqTarget[i]     = random(2, EQ_MAX_TINGGI);
    eqKecepatan[i]  = 0.05f + (random(10) * 0.02f);
    eqWaktuGanti[i] = millis() + random(300, 1200);
  }
  waktuGantiLirik = millis();
}

// ─── Loop ────────────────────────────────────────────────────
void loop() {
  unsigned long sekarang = millis();
  updateLirik(sekarang);
  updateEqualizer(sekarang);
  updateIkon(sekarang);
  if (sekarang - waktuUpdateDisplay >= DISPLAY_INTERVAL) {
    waktuUpdateDisplay = sekarang;
    gambarDisplay();
  }
}

void updateLirik(unsigned long sekarang) {
  if (sekarang - waktuGantiLirik >= lirik[indexLirik].durasi) {
    indexLirik = (indexLirik + 1) % TOTAL_LIRIK;
    waktuGantiLirik = sekarang;
  }
}

void updateEqualizer(unsigned long sekarang) {
  for (int i = 0; i < EQ_BATANG; i++) {
    if (sekarang >= eqWaktuGanti[i]) {
      eqTarget[i]     = random(2, EQ_MAX_TINGGI + 1);
      eqWaktuGanti[i] = sekarang + random(200, 900);
    }
    float selisih = eqTarget[i] - eqSekarang[i];
    if (abs(selisih) > 0.1f)
      eqSekarang[i] += selisih * eqKecepatan[i];
    else
      eqSekarang[i] = eqTarget[i];
  }
}

void updateIkon(unsigned long sekarang) {
  if (sekarang - waktuIkon >= IKON_INTERVAL) {
    ikonTampil = !ikonTampil;
    waktuIkon  = sekarang;
  }
}

void gambarDisplay() {
  display.clearDisplay();
  gambarJudul();
  gambarIkonMusik();
  gambarGarisPemisah();
  gambarLirik();
  gambarEqualizer();
  display.display();
}

void gambarJudul() {
  display.setTextSize(1);
  display.setCursor(2, 0);
  display.print("NOW PLAYING");
}

void gambarIkonMusik() {
  if (!ikonTampil) return;
  int x = 116, y = 1;
  display.drawLine(x+4, y, x+4, y+6, SH110X_WHITE);
  display.fillRect(x, y+5, 4, 3, SH110X_WHITE);
  display.drawLine(x+4, y, x+8, y+2, SH110X_WHITE);
  display.fillRect(x+7, y+6, 3, 3, SH110X_WHITE);
  display.drawLine(x+9, y+2, x+9, y+6, SH110X_WHITE);
}

void gambarGarisPemisah() {
  display.drawLine(0, 10, 127, 10, SH110X_WHITE);
}

void gambarLirik() {
  const char* b1 = lirik[indexLirik].line1;
  const char* b2 = lirik[indexLirik].line2;
  display.setTextSize(1);
  bool ada2 = (b2 != nullptr && b2[0] != '\0');
  if (ada2) {
    display.setCursor(pusatkanTeks(b1), 19); display.print(b1);
    display.setCursor(pusatkanTeks(b2), 30); display.print(b2);
  } else {
    display.setCursor(pusatkanTeks(b1), 27); display.print(b1);
  }
}

int pusatkanTeks(const char* teks) {
  return max((128 - (int)strlen(teks) * 6) / 2, 0);
}

void gambarEqualizer() {
  int totalLebar = (EQ_BATANG * EQ_LEBAR) + ((EQ_BATANG - 1) * EQ_GAP);
  int startX = (128 - totalLebar) / 2;
  for (int i = 0; i < EQ_BATANG; i++) {
    int h = (int)eqSekarang[i]; if (h < 1) h = 1;
    int x = startX + i * (EQ_LEBAR + EQ_GAP);
    display.fillRect(x, EQ_Y_BAWAH - h + 1, EQ_LEBAR, h, SH110X_WHITE);
  }
}

Troubleshooting

🔇 OLED tidak menyala

Cek wiring SDA/SCL. Pastikan pakai 3.3V bukan 5V. Coba jalankan I2C Scanner sketch untuk memastikan alamat 0x3C terdeteksi.

⚡ Upload gagal / port tidak muncul

Tekan tombol BOOT di board, lalu colok USB. Atau hold BOOT, tekan RESET sekali, lepas BOOT. Install driver CH340 jika perlu.

📺 Layar putih / noise

Library salah. Pastikan pakai Adafruit_SH110X bukan library lain. Objek harus Adafruit_SH1106G (bukan SH1107).

🐌 Animasi tersendat

Normal jika ada kode lain yang menggunakan delay(). Pastikan tidak ada delay() besar di luar fungsi yang disediakan.
